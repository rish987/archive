\begin{reference}{rl_theory/note/fnnot}
  \begin{note}
    We use the notation
    $$\fnnot{f}{X}{Y}$$
    to denote a function with domain $X$ and range $Y$;
    that is, 
    $$\text{for all $x \in X$, there is some $y \in Y$ such that $f(x) = y$}.$$%
  \end{note}
\end{reference}
