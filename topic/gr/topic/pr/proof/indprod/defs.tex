   $$
   \CIND{\XV}{\YV}{\ZV} \iff 
   \forall \xyzchoice,
   \CPr{\XV = \XX, \YV = \YY}{\ZV = \ZZ} = 
   \CPr{\XX}{\ZZ}
   \times
   \CPr{\YY}{\ZZ}#1$$%
 }
