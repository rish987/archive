  We have that
  \begin{align*}

    \sum_{\XX \in \TOS{\X}}\sum_{\YY \in \TOS{\Y}}f(\XV = \XX,\YV = \YY)\times \MPRSQ{\SQPP}{\XV = \XX,\YV = \YY}\\

  \end{align*}
\end{proof}
