\begin{proposition}
  \nrp 1
  We have that, \indsymmdefa.%
\end{proposition}

\begin{proof}
  \nrp 2
  We have that
  \begin{align*}
    \CIND{\XV}{\YV}{\ZV} 
    \relr {\iff} {\mrefindprod} &\ 
    \forall \xyzchoice,
    \CPr{\XV = \XX, \YV = \YY}{\ZV = \ZZ} = 
    \CPr{\XX}{\ZZ}
    \times
    \CPr{\YY}{\ZZ}\\
    \relr {\iff} {\mrefindprod} &\ 
    \CIND{\YV}{\XV}{\ZV}.
  \end{align*}%
\end{proof}
