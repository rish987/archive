\def\pathdef#1{%
  for all $\VTXX, \VTXX' \in \VTX^2$,
  $$\PATHGR{\GR}{\VTXX}{\VTXX'} \iff \left\|\PATHSGR{\GR}{\VTXX}{\VTXX'}\right\| > 0$$
  and
  $$\NPATHGR{\GR}{\VTXX}{\VTXX'} \iff \left\|\PATHSGR{\GR}{\VTXX}{\VTXX'}\right\| = 0,$$
  or, for contextual $\GR$,
  $$\PATH{\VTXX}{\VTXX'}$$
  and
  $$\NPATH{\VTXX}{\VTXX'},$$
  respectively#1%
}
