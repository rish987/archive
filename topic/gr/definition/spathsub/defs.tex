\def\spathsubdef#1{%
  for all $\VTX' \subseteq \VTX$ and $\VTXA, \VTXB \in \VTX'^2$,
  $$\SPATHGRSUB{\GR}{\VTX'}{\VTXA}{\VTXB} \iff \left\|\SPATHSGRSUB{\GR}{\VTX'}{\VTXA}{\VTXB}\right\| > 0$$
  and
  $$\SNPATHGRSUB{\GR}{\VTX'}{\VTXA}{\VTXB} \iff \left\|\SPATHSGRSUB{\GR}{\VTX'}{\VTXA}{\VTXB}\right\| = 0,$$
  or, for contextual $\GR$,
  $$\SPATHSUB{\VTX'}{\VTXA}{\VTXB}$$
  and
  $$\SNPATHSUB{\VTX'}{\VTXA}{\VTXB},$$
  respectively#1%
}
