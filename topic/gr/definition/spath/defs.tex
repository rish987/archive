\def\spathdef#1{%
  for all $\VTXA, \VTXB \subseteq \VTX$ such that $\DJ{\VTXA, \VTXB}$,
  $$\SPATHGR{\GR}{\VTXA}{\VTXB} \iff \left\|\SPATHSGR{\GR}{\VTXA}{\VTXB}\right\| > 0$$
  and
  $$\SNPATHGR{\GR}{\VTXA}{\VTXB} \iff \left\|\SPATHSGR{\GR}{\VTXA}{\VTXB}\right\| = 0,$$
  or, for contextual $\GR$,
  $$\SPATH{\VTXA}{\VTXB}$$
  and
  $$\SNPATH{\VTXA}{\VTXB},$$
  respectively#1%
}
