\def\pathsubdef#1{%
  for all $\GRDEF{\GR}{\VTX}{\EDG}$, $\VTX' \subseteq \VTX$ and $\VTXX, \VTXX' \in \VTX'^2$,
  $$\PATHGRSUB{\GR}{\VTX'}{\VTXX}{\VTXX'} \iff \left\|\PATHSGRSUB{\GR}{\VTX'}{\VTXX}{\VTXX'}\right\| > 0$$
  and
  $$\NPATHGRSUB{\GR}{\VTX'}{\VTXX}{\VTXX'} \iff \left\|\PATHSGRSUB{\GR}{\VTX'}{\VTXX}{\VTXX'}\right\| = 0,$$
  or, for contextual $\GRDEF{\GR}{\VTX}{\EDG}$,
  $$\PATHSUB{\VTX'}{\VTXX}{\VTXX'}$$
  and
  $$\NPATHSUB{\VTX'}{\VTXX}{\VTXX'},$$
  respectively#1%
}
