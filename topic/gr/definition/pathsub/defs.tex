\def\pathsubdef#1{%
  for all $\VTXP \subseteq \VTX$ and $\VTXX, \VTXX' \in \VTXP^2$,
  $$\PATHGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'} \iff \left\|\PATHSGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'}\right\| > 0$$
  and
  $$\NPATHGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'} \iff \left\|\PATHSGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'}\right\| = 0,$$
  or, for contextual $\GR$,
  $$\PATHSUB{\VTXP}{\VTXX}{\VTXX'}$$
  and
  $$\NPATHSUB{\VTXP}{\VTXX}{\VTXX'},$$
  respectively#1%
}
