\def\pathsubdef#1{%
  for all $\VTXP \subseteq \VTX$ and $\VTXX, \VTXX' \in {\left(\VTX \setminus \VTXP\right)}^2$,
  $$\PATHGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'} \iff \PATHGR{\GRSUB{\GR}{\VTXP}}{\VTXX}{\VTXX'}$$
  and
  $$\NPATHGRSUB{\GR}{\VTXP}{\VTXX}{\VTXX'} \iff \NPATHGR{\GRSUB{\GR}{\VTXP}}{\VTXX}{\VTXX'},$$
  or, for contextual $\GR$,
  $$\PATHSUB{\VTXP}{\VTXX}{\VTXX'}$$
  and
  $$\NPATHSUB{\VTXP}{\VTXX}{\VTXX'},$$
  respectively#1%
}
