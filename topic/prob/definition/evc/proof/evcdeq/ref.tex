  We have that
  $$
  \ESQC{\DDD}{f}{\YV = \YY}
  \relrg = {\mreflng{\probpath/proof/evsqexpr}} 10pt
  \sum_{\pin\XX\X}\CPRSQ{\DDD}{\XV = \XX}{\YV = \YY} \times f(\XX, \YY)

  \relr = {\mreflng{\probpath/proof/evsqexpr}}
  \ESQC{\DDD}{f}{\YV = \YY}
  $$%
\end{proof}
