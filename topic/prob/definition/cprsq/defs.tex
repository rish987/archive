\def\cprsqdef#1{%
  for \ctx\ $\SSPT{\SPT}{(\XV, \X), (\YV, \Y), (\QV, \Q)}$, and all $\SQQ \in \SQ$ and $\pinw{\XX, \YY \in \X \times \Y}$,
  $$\CPRSQ{\SQQ}{\XV = \XX}{\YV = \YY} = \lim_{\TR \rightarrow \infty}\left(\frac{\sum_{\TR' = 1}^{\TR}\SQIM{\SQQ}{\TR'}{\XV = \XX, \YV = \YY}}{\sum_{\TR' = 1}^{\TR}\SQIM{\SQQ}{\TR'}{\YV = \YY}}\right),$$
  that is, the \scprb\ of \masn\ $\MASN{\XV = \XX}$ in \seq\ $\SQQ$ given \masn\ $\MASN{\YV = \YY}$#1%
}
