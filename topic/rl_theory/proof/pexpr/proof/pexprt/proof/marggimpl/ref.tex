  For random variables

  $$\jpf > 0 \implies \mpf > 0.$$%
\end{proposition}

  $$\Pr{\XX} \relrg = \mreflng{definition}{\margprobpath} 12pt \sum_{\YY' \in \Y}\Pr{\XX, \YY'} \ge \jp \relr > lg 0.$$%
\end{proof}
